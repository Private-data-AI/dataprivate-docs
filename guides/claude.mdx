## Basic pattern

<CodeGroup>
```python Python
import requests
import anthropic

client = anthropic.Anthropic(api_key="sk-ant-...")

def llm_call_safe(user_message: str) -> str:
    # 1. Detect and anonymize
    result = requests.post(
        "https://private-data.ai/api/v1/public/extract",
        headers={"X-API-Key": "your_api_key"},
        json={"text": user_message}
    ).json()

    # 2. Build local entity map
    entity_map = {}
    counters = {}
    for entity in result["entities"]:
        t = entity["type"].upper()
        counters[t] = counters.get(t, 0) + 1
        entity_map[f"[{t}_{counters[t]}]"] = entity["value"]

    # 3. Call Claude with anonymized text
    response = client.messages.create(
        model="claude-sonnet-4-5",
        max_tokens=1024,
        system="Preserve exactly all tokens in the format [TYPE_N] without any modification.",
        messages=[{"role": "user", "content": result["text"]}]
    )

    # 4. Restore locally
    restored = response.content[0].text
    for placeholder, original in entity_map.items():
        restored = restored.replace(placeholder, original)

    return restored

print(llm_call_safe(
    "Summarize this contract for Jean Dupont, IBAN: FR76 3000 6000 0112 3456 7890"
))
```

```javascript JavaScript
import Anthropic from '@anthropic-ai/sdk';

const client = new Anthropic({ apiKey: 'sk-ant-...' });

async function llmCallSafe(userMessage) {
  // 1. Detect and anonymize
  const response = await fetch("https://private-data.ai/api/v1/public/extract", {
    method: "POST",
    headers: { "X-API-Key": "your_api_key", "Content-Type": "application/json" },
    body: JSON.stringify({ text: userMessage })
  });
  const result = await response.json();

  // 2. Build local entity map
  const counters = {};
  const entityMap = {};
  for (const entity of result.entities) {
    const t = entity.type.toUpperCase();
    counters[t] = (counters[t] || 0) + 1;
    entityMap[`[${t}_${counters[t]}]`] = entity.value;
  }

  // 3. Call Claude
  const llmResponse = await client.messages.create({
    model: 'claude-sonnet-4-5',
    max_tokens: 1024,
    system: 'Preserve exactly all tokens in the format [TYPE_N] without any modification.',
    messages: [{ role: 'user', content: result.text }]
  });

  // 4. Restore locally
  let restored = llmResponse.content[0].text;
  for (const [placeholder, original] of Object.entries(entityMap)) {
    restored = restored.replaceAll(placeholder, original);
  }

  return restored;
}

console.log(await llmCallSafe(
  'Summarize this contract for Jean Dupont, IBAN: FR76 3000 6000 0112 3456 7890'
));
```

## With conversation history (multi-turn)

For multi-turn conversations, rebuild the entity map for each new user message and keep the conversation history with anonymized content.

```python Python
import requests
import anthropic

client = anthropic.Anthropic(api_key="sk-ant-...")

SYSTEM_PROMPT = "Preserve exactly all tokens in the format [TYPE_N] without any modification."

def chat_session():
    messages = []

    while True:
        user_input = input("You: ")
        if user_input.lower() == "exit":
            break

        # Anonymize the new message
        result = requests.post(
            "https://private-data.ai/api/v1/public/extract",
            headers={"X-API-Key": "your_api_key"},
            json={"text": user_input}
        ).json()

        # Build entity map for this turn
        entity_map = {}
        counters = {}
        for entity in result["entities"]:
            t = entity["type"].upper()
            counters[t] = counters.get(t, 0) + 1
            entity_map[f"[{t}_{counters[t]}]"] = entity["value"]

        messages.append({"role": "user", "content": result["text"]})

        response = client.messages.create(
            model="claude-sonnet-4-5",
            max_tokens=1024,
            system=SYSTEM_PROMPT,
            messages=messages
        )

        assistant_msg = response.content[0].text
        messages.append({"role": "assistant", "content": assistant_msg})

        # Restore for display
        restored = assistant_msg
        for placeholder, original in entity_map.items():
            restored = restored.replace(placeholder, original)

        print(f"Assistant: {restored}")
```

## With streaming

<Warning>
  Accumulate the full stream before restoring. A placeholder like `[FIRST_NAME_1]` can be split across multiple chunks.
</Warning>

```python Python
import requests
import anthropic

client = anthropic.Anthropic(api_key="sk-ant-...")

result = requests.post(
    "https://private-data.ai/api/v1/public/extract",
    headers={"X-API-Key": "your_api_key"},
    json={"text": "Analyze the file of Marie Martin, born 03/07/1990"}
).json()

# Build entity map
entity_map = {}
counters = {}
for entity in result["entities"]:
    t = entity["type"].upper()
    counters[t] = counters.get(t, 0) + 1
    entity_map[f"[{t}_{counters[t]}]"] = entity["value"]

# Stream
full_response = ""
with client.messages.stream(
    model="claude-sonnet-4-5",
    max_tokens=1024,
    system="Preserve exactly all tokens in the format [TYPE_N].",
    messages=[{"role": "user", "content": result["text"]}]
) as stream:
    for text in stream.text_stream:
        full_response += text
        print(text, end="", flush=True)

# Restore after full stream
restored = full_response
for placeholder, original in entity_map.items():
    restored = restored.replace(placeholder, original)

print(f"\n\nFinal result: {restored}")
```

## FAQ

<AccordionGroup>
  <Accordion title="Does Claude follow token preservation instructions well?">
    Yes, Claude follows formatting instructions reliably. The system prompt `Preserve exactly all tokens in the format [TYPE_N] without any modification` is sufficient in virtually all cases.
  </Accordion>

  <Accordion title="How do I read the response text from Claude's API?">
    The Anthropic API returns `response.content` as a list of blocks. For plain text, `response.content[0].text` is sufficient. If you use tool calls, filter blocks by `type == 'text'`.
  </Accordion>

  <Accordion title="Does this work with Claude Opus and Haiku?">
    Yes, the pattern is identical for all Anthropic models. `claude-haiku-4-5` is the preferred choice if latency is critical.
  </Accordion>

  <Accordion title="What latency does Data Private add?">
    Typically 50–150ms per `/extract` call. To minimize impact, call `/extract` in parallel with any other preparation work before the LLM call.
  </Accordion>
</AccordionGroup>
