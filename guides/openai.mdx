## Basic pattern

<CodeGroup>
```python Python
import requests
import openai

client = openai.OpenAI(api_key="sk-...")

def llm_call_safe(user_message: str) -> str:
    # 1. Detect and anonymize
    result = requests.post(
        "https://private-data.ai/api/v1/public/extract",
        headers={"X-API-Key": "your_api_key"},
        json={"text": user_message}
    ).json()

    # 2. Build local entity map
    entity_map = {}
    counters = {}
    for entity in result["entities"]:
        t = entity["type"].upper()
        counters[t] = counters.get(t, 0) + 1
        entity_map[f"[{t}_{counters[t]}]"] = entity["value"]

    # 3. Call LLM with anonymized text
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {
                "role": "system",
                "content": "Preserve exactly all tokens in the format [TYPE_N] without any modification."
            },
            {"role": "user", "content": result["text"]}
        ]
    )

    # 4. Restore locally
    restored = response.choices[0].message.content
    for placeholder, original in entity_map.items():
        restored = restored.replace(placeholder, original)

    return restored

print(llm_call_safe(
    "Summarize this contract for Jean Dupont, IBAN: FR76 3000 6000 0112 3456 7890"
))
```

```javascript JavaScript
import OpenAI from 'openai';

const openai = new OpenAI({ apiKey: 'sk-...' });

async function llmCallSafe(userMessage) {
  // 1. Detect and anonymize
  const response = await fetch("https://private-data.ai/api/v1/public/extract", {
    method: "POST",
    headers: { "X-API-Key": "your_api_key", "Content-Type": "application/json" },
    body: JSON.stringify({ text: userMessage })
  });
  const result = await response.json();

  // 2. Build local entity map
  const counters = {};
  const entityMap = {};
  for (const entity of result.entities) {
    const t = entity.type.toUpperCase();
    counters[t] = (counters[t] || 0) + 1;
    entityMap[`[${t}_${counters[t]}]`] = entity.value;
  }

  // 3. Call LLM
  const llmResponse = await openai.chat.completions.create({
    model: 'gpt-4o',
    messages: [
      {
        role: 'system',
        content: 'Preserve exactly all tokens in the format [TYPE_N] without any modification.'
      },
      { role: 'user', content: result.text }
    ]
  });

  // 4. Restore locally
  let restored = llmResponse.choices[0].message.content;
  for (const [placeholder, original] of Object.entries(entityMap)) {
    restored = restored.replaceAll(placeholder, original);
  }

  return restored;
}

console.log(await llmCallSafe(
  'Summarize this contract for Jean Dupont, IBAN: FR76 3000 6000 0112 3456 7890'
));
```

## With conversation history (multi-turn)

For multi-turn conversations, rebuild the entity map for each new user message and keep the conversation history with anonymized content.

```python Python
import requests
import openai

client = openai.OpenAI(api_key="sk-...")

SYSTEM_PROMPT = "Preserve exactly all tokens in the format [TYPE_N] without any modification."

def chat_session():
    messages = [{"role": "system", "content": SYSTEM_PROMPT}]

    while True:
        user_input = input("You: ")
        if user_input.lower() == "exit":
            break

        # Anonymize the new message
        result = requests.post(
            "https://private-data.ai/api/v1/public/extract",
            headers={"X-API-Key": "your_api_key"},
            json={"text": user_input}
        ).json()

        # Build entity map for this turn
        entity_map = {}
        counters = {}
        for entity in result["entities"]:
            t = entity["type"].upper()
            counters[t] = counters.get(t, 0) + 1
            entity_map[f"[{t}_{counters[t]}]"] = entity["value"]

        messages.append({"role": "user", "content": result["text"]})

        response = client.chat.completions.create(model="gpt-4o", messages=messages)
        assistant_msg = response.choices[0].message.content
        messages.append({"role": "assistant", "content": assistant_msg})

        # Restore for display
        restored = assistant_msg
        for placeholder, original in entity_map.items():
            restored = restored.replace(placeholder, original)

        print(f"Assistant: {restored}")
```

## With streaming

<Warning>
  Accumulate the full stream before restoring. A placeholder like `[FIRST_NAME_1]` can be split across multiple chunks.
</Warning>

```python Python
import requests
import openai

client = openai.OpenAI(api_key="sk-...")

result = requests.post(
    "https://private-data.ai/api/v1/public/extract",
    headers={"X-API-Key": "your_api_key"},
    json={"text": "Analyze the file of Marie Martin, born 03/07/1990"}
).json()

# Build entity map
entity_map = {}
counters = {}
for entity in result["entities"]:
    t = entity["type"].upper()
    counters[t] = counters.get(t, 0) + 1
    entity_map[f"[{t}_{counters[t]}]"] = entity["value"]

# Stream
full_response = ""
with client.chat.completions.stream(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "Preserve exactly all tokens in the format [TYPE_N]."},
        {"role": "user", "content": result["text"]}
    ]
) as stream:
    for chunk in stream:
        delta = chunk.choices[0].delta.content or ""
        full_response += delta
        print(delta, end="", flush=True)

# Restore after full stream
restored = full_response
for placeholder, original in entity_map.items():
    restored = restored.replace(placeholder, original)

print(f"\n\nFinal result: {restored}")
```

## FAQ

<AccordionGroup>
  <Accordion title="The LLM sometimes modifies my placeholders, what should I do?">
    Add this instruction to your system prompt:
    ```
    Preserve exactly all tokens in the format [TYPE_N] without any modification.
    Do not translate, lowercase, or reformat tokens like [FIRST_NAME_1] or [IBAN_1].
    ```
    If the problem persists, use `gpt-4o` rather than `gpt-3.5-turbo` — smaller models follow formatting instructions less reliably.
  </Accordion>

  <Accordion title="Should I anonymize the system prompt?">
    Only if your system prompt contains PII. In that case, call `/extract` on it separately and build a combined entity map.
  </Accordion>

  <Accordion title="What latency does Data Private add?">
    Typically 50–150ms per `/extract` call. To minimize impact, call `/extract` in parallel with any other preparation work before the LLM call.
  </Accordion>
</AccordionGroup>
